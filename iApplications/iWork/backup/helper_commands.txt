reset the sequence to 1
SELECT SETVAL(
    (SELECT PG_GET_SERIAL_SEQUENCE('"field_data"', 'id')),
    1,
    FALSE);

set the sequence to max(id) + 1
SELECT SETVAL(
    (SELECT PG_GET_SERIAL_SEQUENCE('"field_data"', 'id')),
    (SELECT COALESCE(MAX(id), 1) FROM "public"."field_data"),
    TRUE  -- TRUE indicates that the next value will be max(id) + 1
);
